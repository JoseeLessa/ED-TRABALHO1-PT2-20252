# Nome do executável
TARGET = programa_groner_jose

# Compilador
CC = gcc

# Flags de compilação
CFLAGS = -Wall -Wextra -std=c11 -Iinclude

# Diretórios
SRC_DIR = src
OBJ_DIR = obj

# Arquivos fonte
SRCS = $(SRC_DIR)/main.c \
       $(SRC_DIR)/time.c \
       $(SRC_DIR)/bdtimes.c \
       $(SRC_DIR)/partida.c \
       $(SRC_DIR)/bdpartidas.c \
       $(SRC_DIR)/menu.c

# Arquivos objeto (gerados em obj/)
OBJS = $(SRCS:$(SRC_DIR)/%.c=$(OBJ_DIR)/%.o)

# Regra padrão
all: $(TARGET)

# Linkagem final
$(TARGET): $(OBJS)
	$(CC) $(CFLAGS) -o $@ $^

# Compilação dos .c em .o
$(OBJ_DIR)/%.o: $(SRC_DIR)/%.c
	@mkdir -p $(OBJ_DIR)
	$(CC) $(CFLAGS) -c $< -o $@

# Limpar arquivos gerados
clean:
	rm -rf $(OBJ_DIR) $(TARGET)

# Rodar o programa
run: all
	./$(TARGET)
